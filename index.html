function hitungProject({ hariKerja, tim, }) { // VALIDASI if (hariKerja <= 0) { throw new Error("Hari kerja harus lebih dari 0"); } // ========================= // KONSTANTA // ========================= const TRANSPORT_AWAL_AKHIR = 130000; const TRANSPORT_HARIAN = 30000; const LISTRIK = 100000; const MARGIN = 0.3; // ========================= // HARGA TIM // ========================= const hargaTim = { 1: 455000, 2: 420000, 3: 390000 }; if (!hargaTim[tim]) { throw new Error("Tim tidak valid (pilih 1, 2, atau 3)"); } // ========================= // HITUNG TRANSPORT // ========================= let biayaTransport = 0; if (hariKerja === 1) { biayaTransport = TRANSPORT_AWAL_AKHIR; } else { const hariTengah = hariKerja - 2; biayaTransport = (TRANSPORT_AWAL_AKHIR * 2) + (hariTengah * TRANSPORT_HARIAN); } // ========================= // HITUNG UPAH // ========================= const biayaUpah = hargaTim[tim] * hariKerja; // ========================= // TOTAL MODAL // ========================= const totalBiaya = biayaTransport + biayaUpah + LISTRIK; // ========================= // HARGA JUAL (MARGIN 30%) // ========================= const hargaJual = totalBiaya * (1 + MARGIN); const profit = hargaJual - totalBiaya; // ========================= // OUTPUT // ========================= return { hariKerja, tim, biayaTransport, biayaUpah, biayaListrik: LISTRIK, totalBiaya, hargaJual, profit }; } // ========================= // CONTOH PAKAI // ========================= const hasil = hitungProject({ hariKerja: 7, tim: 1 }); console.log("=== HASIL PERHITUNGAN ==="); console.log(`Hari Kerja : ${hasil.hariKerja}`); console.log(`Tim : ${hasil.tim}`); console.log(`Transport : Rp ${hasil.biayaTransport.toLocaleString()}`); console.log(`Upah Tukang : Rp ${hasil.biayaUpah.toLocaleString()}`); console.log(`Listrik : Rp ${hasil.biayaListrik.toLocaleString()}`); console.log(`Total Modal : Rp ${hasil.totalBiaya.toLocaleString()}`); console.log(`Harga Jual : Rp ${hasil.hargaJual.toLocaleString()}`); console.log(`Profit : Rp ${hasil.profit.toLocaleString()}`);
